Diagrama de Classes UML - Sistema Nodalgrid

Este documento contém a especificação UML da arquitetura atualizada do projeto, refletindo o modelo de Grafo Direcionado Acíclico (DAG) para controle de misturas de commodities, desacoplamento via padrão Observer (Listeners) e herança múltipla de produtos e recipientes.

    %% =====================================================================
    %% INFRAESTRUTURA BLOCKCHAIN & LISTENERS
    %% =====================================================================
    class Blockchain {
        +Bloco ultimo
        +int tamanho
        +dict estado_lotes
        +adicionar_bloco(evento: EventoLogistico, ids_lotes_origem: List~str~) Void
        +rastrear_lote(id_lote: str) List~Dict~
        +validar_integridade_cadeia() bool
        +percorrer_historico(exibir_console: bool) List~Dict~
    }

    class Bloco {
        +EventoLogistico conteudo
        +Bloco bloco_anterior
        +string hash_anterior
        +List~Bloco~ blocos_origem
        +List~string~ hashes_origem
        +string timestamp_bloco
        +string hash_atual
        -_calcular_hash() string
    }

    class ComplianceListener {
        +Blockchain blockchain
        +on_evento_logistico(entidade: EntidadeLogistica, acao: str, **kwargs) Void
    }

    class FabricaEventos {
        <<static>>
        +criar_lote(entidade, lote) EventoColheita
        +criar_transferencia(origem, destino, lote) EventoTransferencia
        +criar_armazenamento(entidade, lote, id_recipiente) EventoArmazenamento
        +criar_fracionamento(entidade, original, extraido, qtd) EventoFracionamento
        +criar_processamento(entidade, lote) EventoProcessamento
        +criar_finalizacao(entidade, lote) EventoFinalizacao
    }

    %% Relacionamentos Criptográficos
    Blockchain "1" o-- "0..*" Bloco : gerencia >
    Bloco "1" o-- "0..1" Bloco : bloco_anterior
    Bloco "1" o-- "0..*" Bloco : blocos_origem (DAG)
    ComplianceListener --> Blockchain : atualiza >
    ComplianceListener --> FabricaEventos : utiliza >

    class ProdutoAgricola {

    class ProdutoPecuario {

    %% =====================================================================
    %% EVENTOS LOGÍSTICOS & MOTOR DE RISCOS
    %% =====================================================================
    class EventoLogistico {
        <<abstract>>
        +UUID id_evento
        +string timestamp
        +EntidadeLogistica entidade_responsavel
        +LoteAgro lote_envolvido
        +bool selo_procedencia
        +ValidadorRisco validador
        +validar_regras()* bool
        +executar_acao()* Void
        +resumir_evento() dict
        +exibir_resumo() Void
    }

    class EventoColheita {
        +validar_regras() bool
        +executar_acao() Void
    }

    class EventoArmazenamento {
        +string id_recipiente
        +validar_regras() bool
        +executar_acao() Void
    }

    class EventoTransferencia {
        +EntidadeLogistica destino
        +validar_regras() bool
        +executar_acao() Void
    }

    class EventoFinalizacao {
        +validar_regras() bool
        +executar_acao() Void
    }

    class ValidadorRisco {
        -_MAPEAMENTO_RISCOS : dict
        +validar(evento: EventoLogistico, tipo_risco: str, deve_falhar: bool, mensagem: str) bool
    }

    %% Relacionamentos de Eventos
    EventoLogistico <|-- EventoColheita
    EventoLogistico <|-- EventoArmazenamento
    EventoLogistico <|-- EventoTransferencia
    EventoLogistico <|-- EventoFinalizacao

    EventoLogistico --> ValidadorRisco : usa para auditoria >
    Bloco --> EventoLogistico : carrega payload >
    EventoLogistico --> EntidadeLogistica : referencia >
